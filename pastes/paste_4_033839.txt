    <script>
        const circumference = 2 * Math.PI * 65;
        
        function setGauge(id, percent) {
            const gauge = document.getElementById(id);
            const offset = circumference - (percent / 100) * circumference;
            gauge.style.strokeDashoffset = offset;
            
            if (percent < 50) {
                gauge.style.stroke = '#3fb950';
            } else if (percent < 80) {
                gauge.style.stroke = '#d29922';
            } else {
                gauge.style.stroke = '#f85149';
            }
        }
        
        function setGaugeValue(valueId, percent) {
            document.getElementById(valueId).textContent = percent;
        }
        
        async function updateMetrics() {
            try {
                const data = JSON.parse(await fetch('/resources/api/metrics').then(r => r.text()));
                
                setGauge('cpu-gauge', data.cpu.usage);
                setGaugeValue('cpu-value', data.cpu.usage);
                document.getElementById('cpu-cores').textContent = data.cpu.cores + ' cores';
                
                const maxLoad = Math.max(data.load.load1, data.load.load5, data.load.load15, 1);
                document.getElementById('load-1').style.height = (data.load.load1 / maxLoad * 100) + '%';
                document.getElementById('load-5').style.height = (data.load.load5 / maxLoad * 100) + '%';
                document.getElementById('load-15').style.height = (data.load.load15 / maxLoad * 100) + '%';
                
                setGauge('ram-gauge', data.ram.percent);
                setGaugeValue('ram-value', data.ram.percent);
                document.getElementById('ram-used').textContent = data.ram.used;
                document.getElementById('ram-total').textContent = data.ram.total;
                
                setGauge('disk-gauge', data.disk.percent);
                setGaugeValue('disk-value', data.disk.percent);
                document.getElementById('disk-used').textContent = data.disk.used;
                document.getElementById('disk-total').textContent = data.disk.total;
                
                document.getElementById('uptime-days').textContent = data.uptime.days;
                document.getElementById('uptime-hours').textContent = data.uptime.hours;
                document.getElementById('uptime-mins').textContent = data.uptime.mins;
                
                let servicesHtml = '';
                data.services.forEach(svc => {
                    const dotClass = svc.status === 'active' ? '' : 'inactive';
                    servicesHtml += '<div class="service-item">' +
                        '<span class="service-name">' + svc.name + '</span>' +
                        '<span class="service-status">' +
                            '<span class="status-dot ' + dotClass + '"></span>' +
                            svc.status +
                        '</span>' +
                    '</div>';
                });
                document.getElementById('services-list').innerHTML = servicesHtml;
                
                if (data.network && data.network.ip) {
                    document.getElementById('network-ip').textContent = data.network.ip;
                }
                
            } catch(e) {
                console.error('Error fetching metrics:', e);
            }
        }
        
        setGauge('cpu-gauge', 0);
        setGauge('ram-gauge', 0);
        setGauge('disk-gauge', 0);
        
        updateMetrics();
        setInterval(updateMetrics, 1000);
    </script>
